class Solution {
public:
    int findMin(vector<int>& nums) {
        int l=0,r=nums.size()-1;
        int last=nums.back();
        int ans=INT_MAX;
        while(l<=r){
            int mid=l+r>>1;
            if(nums[mid]>last){
               l=mid+1;
            }else{
                r=mid-1;
            }
            ans=min(ans,nums[mid]);
        }
        return ans;
    }
};
