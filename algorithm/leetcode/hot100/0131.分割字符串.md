分割成回文，对第i个字母不断进行搜索+回溯
```cpp
class Solution {
public:
    bool check(const string &s,int l,int r){
        while(l<=r){
            if(s[l++]!=s[r--])return false;
        }
        return true;
    }
    vector<vector<string>> partition(string s) {
        vector<vector<string>>ans;
        vector<string>t;
        int n=s.size();
        auto dfs=[&](this auto&&dfs,int i){
            if(i==n){
                ans.push_back(t);
                return ;
            }
            for(int j=i;j<n;j++){
                if(check(s,i,j)){
                    t.push_back(s.substr(i,j-i+1));
                    dfs(j+1);
                    t.pop_back();
                }
            }
        };
        dfs(0);
        return ans;
    }
};
