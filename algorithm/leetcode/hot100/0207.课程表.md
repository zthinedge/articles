class Solution {
public:
    bool canFinish(int numCourses, vector<vector<int>>& prerequisites) {
        int n=numCourses;
        vector<int>ind(n);
        vector<vector<int>>g(n);
        for(int i=0;i<prerequisites.size();i++){
            int a=prerequisites[i][0],b=prerequisites[i][1];
            ind[a]++;
            g[b].push_back(a);
        }
        queue<int>q;
        for(int i=0;i<n;i++){
            if(ind[i]==0)q.push(i);
        }
        int lesson=0;
        while(!q.empty()){
            int t=q.front();
            q.pop();
            lesson++;
            for(int i=0;i<g[t].size();i++){
                if(--ind[g[t][i]]==0){
                    q.push(g[t][i]);
                }
            }
        }
        return lesson==n;
    }
};
