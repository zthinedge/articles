
就是我要对每一个字母依次进行组合，搜索的终止条件是当前字符串长度等于digits长度。根据当前字符串的长度可以作为下标访问号码组合
```cpp
class Solution {
public:
    vector<string>ans;
    void dfs(string&digits,vector<string>&mp,string&t){
        if(digits.size()==t.size()){
            ans.push_back(t);
            return ;
        }
        char key=digits[t.size()];
        for(int i=0;i<mp[key-'2'].size();i++){
            t.push_back(mp[key-'2'][i]);
            dfs(digits,mp,t);
            t.pop_back();
        }
    }
    vector<string> letterCombinations(string digits) {
        vector<string>mp(8);
        string t="";
        mp={"abc","def","ghi","jkl","mno","pqrs","tuv","wxyz"};
        dfs(digits,mp,t);
        return ans;
    }
};
