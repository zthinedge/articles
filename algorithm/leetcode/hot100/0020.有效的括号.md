class Solution {
public:
    char check(char c){
        if(c=='(')return ')';
        if(c=='[')return ']';
        if(c=='{')return '}';
        return 0;
    }
    bool isValid(string s) {
        stack<char>st;
        for(int i=0;i<s.size();i++){
            if(!st.empty()&&check(st.top())==s[i]){
                st.pop();
            }else{
                st.push(s[i]);
            }
        }
        return st.empty();
    }
};
