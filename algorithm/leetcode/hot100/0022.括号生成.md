考虑搜索的边界条件未当前字符串大小达到2*n的时候。然后每一种括号需要有n个，左括号要大于等于右括号才行。所以优先加左括号回溯，再加右括号
```cpp
class Solution {
public:
    string t;
    vector<string>ans;
    void dfs(string &t,int n,int l,int r){
        if(t.size()==2*n){
            ans.push_back(t);
            return ;
        }
        if(l<n){
            t.push_back('(');
            dfs(t,n,l+1,r);
            t.pop_back();
        }
        if(r<l){
            t.push_back(')');
            dfs(t,n,l,r+1);
            t.pop_back();
        }

    }
    vector<string> generateParenthesis(int n) {
        t="";
        dfs(t,n,0,0);
        return ans;    
    }
};
