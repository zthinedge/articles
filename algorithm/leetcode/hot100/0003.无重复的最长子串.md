class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int cnt[256]={0};
        int n=s.size();
        int ans=0;
        for(int i=0,j=0;j<n;j++){
            if(++cnt[s[j]]==1){
                ans=max(ans,j-i+1);
                continue;
            }
            while(cnt[s[j]]>1){
                cnt[s[i]]--;
                i++;
            }
        }
        return ans;
    }
};
