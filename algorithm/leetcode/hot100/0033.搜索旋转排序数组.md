题目要求用O(logn) 来解决问题，所以要采用二分的方式，因此需要设计二分边界。最初始是单调递增的 ，调整后会出现两端增加的序列，同时``last<nums[0]``，每次取到中间值，需要判断它的位置
```cpp
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int l=0,r=nums.size()-1;
        int last=nums.back();
        while(l<=r){
            int mid=l+r>>1;
            if(nums[mid]==target)return mid;
            else if(nums[mid]>last){
                if(target>last&&target<nums[mid]){
                    r=mid-1;
                }else{
                   l=mid+1;
                }
            }else{
                if(target<=last&&target>nums[mid]){
                    l=mid+1;
                }else{
                    r=mid-1;
                }
            }
        }
        return -1; 
    }
};
