class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int n=nums.size();
        map<int,int>mp;
        int ans=-1,maxn=0;
        for(auto num:nums){
            mp[num]++;
            if(mp[num]>maxn){
                maxn=mp[num];
                ans=num;
            }
        }
        return ans;
    }
};
