题目需要返回橘子全部腐烂的最小分钟数，那么可以先把已经腐烂的橘子放入队列以便后续的扩散，同时记录还没有被腐烂的橘子数目。然后对其进行BFS,每次弹出队列中所有腐烂的橘子并进行一层扩散，将新鲜的橘子标记为腐烂放入队列。然后时间+1，当队列为空或者说橘子已经全部腐烂时退出循环，再根据橘子是否全部腐烂返回对应结果
```cpp
class Solution {
public:
    struct node{
        int x,y;
    };
    int dir[4][2]={{0,1},{1,0},{-1,0},{0,-1}};
    int orangesRotting(vector<vector<int>>& grid) {
        int m=grid.size();
        int n=grid[0].size();
        queue<node>q;
        int num=0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(grid[i][j]==2){
                    q.push({i,j});
                }
                if(grid[i][j]==1)num++;
            }
        }
        int ans=0;
        while(num&&!q.empty()){
            int sz=q.size();
            for(int i=0;i<sz;i++){
                auto t=q.front();
                q.pop();
                for(int i=0;i<4;i++){
                    int nx=t.x+dir[i][0],ny=t.y+dir[i][1];
                    if(nx>=0&&nx<m&&ny>=0&&ny<n&&grid[nx][ny]==1){
                        q.push({nx,ny});
                        grid[nx][ny]=2;
                        num--;
                    }
                }
            } ans++;
            
        }
        return num?-1:ans;
    }
};
