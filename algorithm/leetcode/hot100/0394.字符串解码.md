此题需要注意的细节很多，比如对于栈，需要先压入数字和字母，遇到``]``进行结算，中间可能会会忽略边界
```cpp
class Solution {
public:
    string getDigits(string &s,int &p){
        string res="";
        while(isdigit(s[p])){
            res.push_back(s[p++]);
        }
        return res;
    }
     string getString(vector <string> &v) {
        string ret;
        for (const auto &s: v) {
            ret += s;
        }
        return ret;
    }

    string decodeString(string s) {
        vector<string>stk;
        int p=0;
        while(p<s.size()){
            if(isdigit(s[p])){
                string digits=getDigits(s,p);
                stk.push_back(digits);
            }else if(isalpha(s[p])||s[p]=='['){
                stk.push_back(string(1, s[p]));
                p++;
            }else{
                p++;
                vector<string>sub;
                while(stk.back()!="["){
                    sub.push_back(stk.back());
                    stk.pop_back();
                }
                reverse(sub.begin(),sub.end());
                stk.pop_back();
                int t=stoi(stk.back());
                stk.pop_back();
                string ss,o=getString(sub);
                while(t--)ss+=o;
                stk.push_back(ss);
            }
        }
        return getString(stk);
    }
};
