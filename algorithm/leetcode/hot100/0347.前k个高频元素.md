class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        unordered_map<int,int>cnt;
        int max_cnt=0;
        for(int num:nums){
            cnt[num]++;
            max_cnt=max(cnt[num],max_cnt);
        }
        vector<vector<int>>t(max_cnt+1);
        for(auto&[k,v]:cnt){
            t[v].push_back(k);
        }
        vector<int>ans;
        for(int i=max_cnt;i>=0&&ans.size()<k;i--){
             ans.insert(ans.end(),t[i].begin(),t[i].end());
        }
        return ans;
    }
};
