维护一个最小栈，需要把每一次push进栈中的元素如果<=栈顶，就压入最小栈
```cpp
class MinStack {
public:
    stack<int>s;
    stack<int>ms;
    MinStack() {
       ms.push(INT_MAX); 
    }
    
    void push(int val) {
        s.push(val);
        if(val<=ms.top())ms.push(val);
    }
    
    void pop() {
        if(ms.top()==s.top())ms.pop();
        s.pop();
    }
    
    int top() {
        return s.top();
    }
    
    int getMin() {
        return ms.top();
    }
};

/**
 * Your MinStack object will be instantiated and called as such:
 * MinStack* obj = new MinStack();
 * obj->push(val);
 * obj->pop();
 * int param_3 = obj->top();
 * int param_4 = obj->getMin();
 */
