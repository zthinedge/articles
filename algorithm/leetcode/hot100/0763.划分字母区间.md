先记录每一个字母最后出现的位置，然后遍历字符串，如果达到边界就进行一次结算
```cpp
class Solution {
public:
    vector<int> partitionLabels(string s) {
        int n=s.size();
        vector<int>ans;
        int step[26];
        for(int i=0;i<n;i++){
            step[s[i]-'a']=i;
        }
        int maxn=0;
        int pre=-1;
        for(int i=0;i<n;i++){
            maxn=max(maxn,step[s[i]-'a']);
            if(maxn==i){
                ans.push_back(i-pre);
                pre=i;
            } 
        }
        return ans;
    }
};
