class Solution {
public:
    vector<vector<int>> permute(vector<int>& nums) {
        int n=nums.size();
        vector<vector<int>>ans;
        vector<int>path(n),vis(n);
        auto dfs=[&](this auto&&dfs,int step){
            if(step==n){
                ans.push_back(path);
                return ;
            }
            for(int i=0;i<n;i++){
                if(!vis[i]){
                    vis[i]=1;
                    path[step]=nums[i];
                    dfs(step+1);
                    vis[i]=0;
                }
            }

        };
        dfs(0);
        return ans;
    }
};
