要求柱状图中最大的矩形的话，肯定要求对于每一个高度，找到它左右第一个比他小的数。构造一个单调递增栈

```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int n=heights.size();
        int ans=0;
        stack<int>st;
        vector<int>l(n),r(n,n);
        for(int i=0;i<n;i++){
            while(!st.empty()&&heights[st.top()]>=heights[i]){
                r[st.top()]=i;
                st.pop();
            }
            l[i]=st.empty()?-1:st.top();
            st.push(i);
        }
        for(int i=0;i<n;i++){
            ans=max(ans,(r[i]-l[i]-1)*heights[i]);
        }
        return ans;
    }
};
