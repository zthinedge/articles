因为数组中有1+n个数，有一个数字重复，构造`i->nums[i]`的边，一定会存在一个节点入度为2 构成环，然后可以通过快慢指针来进行求解
```cpp
class Solution {
public:
    int findDuplicate(vector<int>& nums) {
        int slow=0,fast=0;
        while(1){
            slow=nums[slow];
            fast=nums[nums[fast]];
            if(slow==fast)break;
        }
        int head=0;
        while(head!=slow){
            slow=nums[slow];
            head=nums[head];
        }
        return head;
    }
};
