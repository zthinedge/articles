题目要求跳到n-1位置的最小步数。一旦达到当前范围的边界，就需要再跳一步，并更新边界为之前所能记录跳到的最大值。
```cpp
class Solution {
public:
    int jump(vector<int>& nums) {
        int n=nums.size();
        int step=0;
        int end=0;
        int maxn=0;
        for(int i=0;i<n-1;i++){
            maxn=max(maxn,i+nums[i]);
            if(i==end){
                step++;
                end=max(end,maxn);
            }
        }
        return step;
    }
};
