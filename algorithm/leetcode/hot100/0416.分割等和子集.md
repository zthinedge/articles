class Solution {
public:
    bool canPartition(vector<int>& nums) {
        // 1. 计算数组总和
        int sum = accumulate(nums.begin(), nums.end(), 0);
        // 总和为奇数，直接返回false（无法平分）
        if (sum % 2 != 0) {
            return false;
        }

        // 2. 定义目标值 + 提前判断最大元素
        int target = sum / 2;
        int max_num = *max_element(nums.begin(), nums.end());
        if (max_num > target) {
            return false;
        }

        // 3. 一维DP数组初始化（dp[j]表示是否存在和为j的子集）
        vector<bool> dp(target + 1, false);
        dp[0] = true; // 空集和为0，恒成立

        // 4. 0-1背包核心逻辑（倒序遍历避免重复选元素）
        for (int num : nums) {
            for (int j = target; j >= num; --j) {
                // 选当前元素：dp[j-num]；不选：dp[j]；只要一个为true则dp[j]为true
                dp[j] = dp[j] || dp[j - num];
            }
        }

        // 5. 返回是否存在和为target的子集
        return dp[target];
    }
};
