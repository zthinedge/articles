双指针+dp
dp[i]表示，前i-1个节点满足的最小长度
用双指针找到满足sum==target的条件后，维护res为当前长度和之前长度的和的最小值
```cpp
class Solution {
public:
    int minSumOfLengths(vector<int>& arr, int target) {
        int n=arr.size();
        vector<int>dp(n+1);
        dp[0]=1e8;
        int res=INT_MAX;
        int sum=0;
        for(int i=0,j=0;j<n;j++){
            sum+=arr[j];
            while(sum>target){
                sum-=arr[i++];
            }
            if(sum==target){
                int len=j-i+1;
                res=min(res,len+dp[i]);
                dp[j+1]=min(dp[j],len);
            }else{
                dp[j+1]=dp[j];
            }
        }
        return res>n?-1:res;
    }
};
