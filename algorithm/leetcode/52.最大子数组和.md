用一个ans来更新sum的最大值，sum为以第i个数结尾的最大和
```cpp
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int sum=nums[0];
        int ans=sum;
        for(int i=1;i<nums.size();i++){
            sum=max(nums[i],sum+nums[i]);
            ans=max(sum,ans);
        }
        return ans;
    }
};
