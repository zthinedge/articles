class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        int m=matrix.size();
        int n=matrix[0].size();
        vector<vector<int>>vis(m,vector<int>(n));
        vector<int>res;
        int tot=0;
        int x=0,y=0;
        vis[x][y]=1;
        res.push_back(matrix[x][y]);
        tot++;
        while(tot<m*n){
            while(y+1<n&&!vis[x][y+1]){
                y++;
                res.push_back(matrix[x][y]);
                vis[x][y]=1;
                tot++;
            }
            while(x+1<m&&!vis[x+1][y]){
                x++;
                res.push_back(matrix[x][y]);
                vis[x][y]=1;
                tot++;
            }
            while(y-1>=0&&!vis[x][y-1]){
                y--;
                res.push_back(matrix[x][y]);
                vis[x][y]=1;
                tot++;
            }
            while(x-1>=0&&!vis[x-1][y]){
                x--;
                res.push_back(matrix[x][y]);
                vis[x][y]=1;
                tot++;
            }
        }
        return res;
    }
};
