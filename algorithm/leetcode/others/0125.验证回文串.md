class Solution {
public:
    bool isPalindrome(string s) {
        int l=0,r=s.size()-1;
        for(int i=0;i<s.size();i++){
            if(s[i]>='A'&&s[i]<='Z'){
                s[i]+='a'-'A';
            }
        }
        while(l<=r){
            while (l <= r && !isalnum(s[l])) l++; 
            while (l <= r && !isalnum(s[r])) r--;
            if (l <= r && s[l++] != s[r--]) {
                return false;
            }
        }
        return true;
    }
};
