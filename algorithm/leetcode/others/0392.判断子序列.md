class Solution {
public:
    bool isSubsequence(string s, string t) {
        if(s.size()>t.size())return false;
        int p1=0,p2=0;
        for(p2;p2<t.size()&&p1<s.size();p2++){
            if(s[p1]==t[p2]){
                p1++;
            }
        }
        return p1==s.size();
    }
};
