class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        unordered_set<int>s(nums.begin(),nums.end());
        int ans=0;
        for(auto i:s){
            if(s.contains(i-1)){
                continue;
            }
            int j=i+1;
            while(s.contains(j)){
                j++;
            }
            ans=max(ans,j-i);
        }
        return ans;
    }
};
