
`dp[i][j]`表示在`i~j`区间内是否为回文串，用`max_len`维护最长长度，时间复杂度为`O(n^2)`
```cpp
class Solution {
public:
    string longestPalindrome(string s) {
        int n=s.size();
        vector<vector<int>>dp(n,vector<int>(n));
        int max_len=1;
        int begin=0;
        for(int i=0;i<n;i++)dp[i][i]=1;
        for(int L=2;L<=n;L++){
            for(int i=0;i<n;i++){
                int j=i+L-1;
                if(j>=n)break;
                if(s[i]!=s[j])dp[i][j]=0;
                else{
                    if(j-i<3)dp[i][j]=1;
                    else dp[i][j]=dp[i+1][j-1];
                }
                if(dp[i][j]&&j-i+1>max_len){
                    max_len=j-i+1;
                    begin=i;
                }
            }
        }
        return s.substr(begin,max_len);
    }
};
