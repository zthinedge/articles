
先随便举例一个数789321，它的下一个数是791238，我们可以发现规律，先要从右往左找到第一个 `nums[i]<nums[i+1]` 的数字，比如这里找到的是8 ，然后找到一个数使得这个数要比8要大，然后将其进行交换，对后面的数字整体进行翻转。
```cpp
class Solution {
public:
    void nextPermutation(vector<int>& nums) {
        int n=nums.size();
        int i=n-2;
        while(i>=0&&nums[i]>=nums[i+1]){
            i--;
        }
        int j=n-1;
        if(i>=0){
            while(nums[j]<=nums[i]){
                j--;
            }
            swap(nums[j],nums[i]);
        }
        reverse(nums.begin()+i+1,nums.end());
        
    }
};
