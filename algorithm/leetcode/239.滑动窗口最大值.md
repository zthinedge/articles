用单调队列实现，思路是维护一个定长区间内的最大值，将每一个元素的下标push进双端队列中，如果新来的元素大于尾部元素，说明这个尾部元素之后不会再是最大值了，将尾部元素弹出。总之是求最大值维护一个单调递减的队列，队首就是最大值
```cpp
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        vector<int>ans;
        deque<int>q;
        int n=nums.size();
        for(int i=0;i<n;i++){
            if(i>=k&&i-k+1>q.front())q.pop_front();
            while(!q.empty()&&nums[q.back()]<=nums[i]){
                q.pop_back();
            }
            q.push_back(i);
            if(i>=k-1)ans.push_back(nums[q.front()]);
        }
        return ans;
    }
};
