class Solution {
public:
    int nextGreaterElement(int n) {
       string s=to_string(n);
       int sz=s.size();
       int i=sz-2;
       while(i>=0&&s[i]>=s[i+1]){
        i--;
       } 
       if(i==-1)return -1;
       int j=sz-1;
       while(s[j]<=s[i]){
        j--;
       }
       swap(s[j],s[i]);
       reverse(s.begin()+i+1,s.end());
       long ans=stol(s);
       return ans>INT_MAX?-1:ans;
    }
};
