class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        vector<int>ans;
        vector<int>ss(26),pp(26);
        for(int i=0;i<p.size();i++){
            pp[p[i]-'a']++;
        }
        for(int i=0,j=0;i<s.size();i++){
            ss[s[i]-'a']++;
            
            while(ss[s[i]-'a']>pp[s[i]-'a']){
                ss[s[j]-'a']--;
                j++;
            }
            if(ss==pp)ans.push_back(j);
        }
        return ans;
    }
};
