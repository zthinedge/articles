采用双指针实现，每一个点i所接到的雨水量为min(左边的最高,右边的最高)-当前地面高度
```cpp
class Solution {
public:
    int trap(vector<int>& height) {
        int n=height.size();
        int l=0,r=n-1,ans=0,lm=0,rm=0;
        while(l<r){
            lm=max(height[l],lm);
            rm=max(height[r],rm);
            if(lm<rm){
                ans+=lm-height[l++];
            }else{
                ans+=rm-height[r--];
            }
        }
        return ans;
    }
};
