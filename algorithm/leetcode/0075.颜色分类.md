
想办法把`0`换到前面，`2`换到后面，用双指针进行处理。对于`nums[i]`为`2 `，它要与后面的数交换，可能换回来是`2，1，0`，需要换到当前数字不再为`2`为止。然后如果最后换到的是`0`，将`0`与前面的`l`指针进行交换，`l++`
```cpp
class Solution {
public:
    void sortColors(vector<int>& nums) {
        int n=nums.size();
        int l=0,r=n-1;
        for(int i=0;i<=r;i++){
            while(i<=r&&nums[i]==2){
                swap(nums[i],nums[r--]);
            }if(nums[i]==0){
                swap(nums[i],nums[l++]);
            }
        }
    }
};
