思路为先欠下多少个字符，之后再还清缩减长度。
```cpp
class Solution {
public:
    string minWindow(string s, string t) {
        int cnt[256]={0};
        int minlen=1e6,start=0;
        int debt=t.size();
        for(int i=0;i<debt;i++)cnt[t[i]]--;
        for(int i=0,j=0;j<s.size();j++){
            if(cnt[s[j]]++<0){
                debt--;
            }
            if(debt==0){
                while(cnt[s[i]]>0){
                    cnt[s[i++]]--;
                }
                int len=j-i+1;
                if(len<minlen){
                    minlen=len;
                    start=i;
                }
            }
        }
        return minlen>1e5?"":s.substr(start,minlen);
    }
};
