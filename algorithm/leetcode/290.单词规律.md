要保证pattern和s是双向映射，使用哈希表，同时要保证对应的长度相等
class Solution {
public:
    bool wordPattern(string pattern, string s) {
        unordered_map<string,char>s2p;
        unordered_map<char,string>p2s;
        int n=pattern.size();
        int i=0;
        string str;
        for(int j=0;j<s.size();j++){
            if(s[j]!=' ')str+=s[j];
            if(s[j]==' '||j==s.size()-1){
                if(i>=n)return false;
                char ch=pattern[i];
                if(!p2s.count(ch)&&!s2p.count(str)){
                    p2s[ch]=str;
                    s2p[str]=ch;
                }else{
                    if(p2s[ch]!=str||s2p[str]!=ch){
                        return false;
                    }
                }
                str="";
                i++;
            }
        }
        return i==n;
    }
};
