class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int n=nums.size();
        int k=0;
        int l=0,r=0;
        while(r<n){
            if(nums[r]!=val){
                swap(nums[l],nums[r]);
                l++;
            }
            r++;
        }
        return l;
    }
};
